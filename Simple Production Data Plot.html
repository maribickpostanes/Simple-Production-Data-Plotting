# Import python libraries
import matplotlib.pyplot as plt
import pandas as pd

# Read Excel File
df = pd.read_excel('volve.xlsx')

# Create lists and filter the data frame to retrieve only two wells
x = df[df["Wellbore name"]=='15/9-F-12 H']
y = df[df["Wellbore name"]=='15/9-F-14 H']

# Plot the data and do some customization
fig, ax = plt.subplots(figsize=[9,6])
x.plot(x='Date of Production',y='BORE_OIL_VOL',c='#425df5',label='NO 15/9-F-12 H',ax=ax)
y.plot(x='Date of Production',y='BORE_OIL_VOL',ylabel='Borehole Oil Volume\n(Sm3)',c='orange',label='NO 15/9-F-14 H',ax=ax)
plt.grid(True);plt.legend(title='Wellbore Name')
plt.title('DAILY OIL PRODUCTION',fontsize=15)
